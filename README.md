To install Rofi, install the following dependencies:

```
sudo apt update && sudo apt install -y \
  build-essential \
  autoconf \
  automake \
  pkg-config \
  flex \
  bison \
  check \
  libpango1.0-dev \
  libcairo2-dev \
  libglib2.0-dev \
  libgdk-pixbuf2.0-dev \
  libstartup-notification0-dev \
  libxkbcommon-dev \
  libxkbcommon-x11-dev \
  libxcb1-dev \
  libxcb-xkb-dev \
  libxcb-randr0-dev \
  libxcb-xinerama0-dev \
  libxcb-util-dev \
  libxcb-ewmh-dev \
  libxcb-icccm4-dev \
  libxcb-cursor-dev \
  xcb-proto \
  libxcb-imdkit-dev
```
